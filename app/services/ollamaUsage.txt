const timing = (start) => `${((Date.now() - start) / 1000).toFixed(2)}s`;

}

async function test1_generateText() {

    console.log(`  ⏱️  Time      : ${timing(t)}`);

      pass("generateText returns text + stop");

  } catch (e) {
    fail("generateText", e.message);
  }

// ───────────────────────────────────────────────────────────────────────────
// 2. streamText — streaming chunks
// ───────────────────────────────────────────────────────────────────────────
async function test2_streamText() {
  header(2, "streamText — Streaming delta chunks");
  const t = Date.now();
  try {
    const result = streamText({
      model,
      prompt: "Đếm từ 1 đến 5, mỗi số một dòng. Không thêm gì khác.",
    });

    process.stdout.write("  📡 Stream    : ");
    let fullText = "",
      chunkCount = 0;
    for await (const chunk of result.textStream) {
      process.stdout.write(chunk);
      fullText += chunk;
      chunkCount++;
    }

    const usage = await result.usage;
    const finishReason = await result.finishReason;
    console.log(`\n  🧩 Chunks    : ${chunkCount}`);
    console.log(`  🏁 Finish    : ${finishReason}`);
    console.log(
      `  📊 Tokens    : in=${usage.promptTokens} out=${usage.completionTokens}`,
    );
    console.log(`  ⏱️  Time      : ${timing(t)}`);

    if (chunkCount > 1) pass(`streamText — ${chunkCount} chunks received`);
    else fail("streamText", "No chunks or single chunk only");
  } catch (e) {
    fail("streamText", e.message);
  }
}

// ───────────────────────────────────────────────────────────────────────────
// 3. generateObject — structured Zod schema
// ───────────────────────────────────────────────────────────────────────────
async function test3_generateObject() {
  header(3, "generateObject — Structured JSON with Zod");
  const t = Date.now();

  const PersonSchema = z.object({
    name: z.string().describe("Tên nhân vật"),
    age: z.number().describe("Tuổi"),
    occupation: z.string().describe("Nghề nghiệp"),
    skills: z.array(z.string()).describe("Danh sách kỹ năng"),
    isHero: z.boolean().describe("Có phải nhân vật chính không"),
  });

  try {
    const { object, usage } = await generateObject({
      model,
      schema: PersonSchema,
      prompt: "Tạo thông tin cho một nhân vật hư cấu là lập trình viên AI.",
    });

    console.log(
      "  📦 Object    :",
      JSON.stringify(object, null, 4).replace(/\n/g, "\n               "),
    );
    console.log(
      `  📊 Tokens    : in=${usage.promptTokens} out=${usage.completionTokens}`,
    );
    console.log(`  ⏱️  Time      : ${timing(t)}`);

    const parsed = PersonSchema.safeParse(object);
    if (parsed.success) pass("generateObject — Zod validation passed");
    else fail("generateObject schema", parsed.error.message);
  } catch (e) {
    fail("generateObject", e.message);
  }
}

// ───────────────────────────────────────────────────────────────────────────
// 4. streamObject — streaming structured output
// ───────────────────────────────────────────────────────────────────────────
async function test4_streamObject() {
  header(4, "streamObject — Streaming structured JSON");
  const t = Date.now();

  const ProductSchema = z.object({
    productName: z.string(),
    price: z.number(),
    category: z.string(),
    features: z.array(z.string()),
    inStock: z.boolean(),
  });

  try {
    const result = streamObject({
      model,
      schema: ProductSchema,
      prompt: "Tạo thông tin cho một chiếc laptop gaming cao cấp.",
    });

    process.stdout.write("  📡 Streaming : ");
    for await (const _ of result.partialObjectStream) process.stdout.write(".");
    console.log("");

    const finalObj = await result.object;
    console.log(
      "  📦 Final     :",
      JSON.stringify(finalObj, null, 4).replace(/\n/g, "\n               "),
    );
    console.log(`  ⏱️  Time      : ${timing(t)}`);

    const parsed = ProductSchema.safeParse(finalObj);
    if (parsed.success)
      pass("streamObject — final object passes Zod validation");
    else fail("streamObject", parsed.error.message);
  } catch (e) {
    fail("streamObject", e.message);
  }
}

// ───────────────────────────────────────────────────────────────────────────
// 5. Tool Calling — generateText via OpenAI-compat endpoint
//    Lý do dùng modelOAI: ollama-ai-provider v0.15 không forward tools
//    xuống Ollama API đúng cách. OpenAI-compat endpoint hoạt động tốt.
// ───────────────────────────────────────────────────────────────────────────
async function test5_toolCalling() {
  header(5, "Tool Calling — generateText + tools (OpenAI-compat)");
  const t = Date.now();
  try {
    const { text, toolCalls, toolResults, steps, finishReason } =
      await generateText({
        model,
        tools,
        maxSteps: 5,
        prompt:
          "Thời tiết Hà Nội và TP.HCM hôm nay thế nào? Và tính giúp tôi 15 * 8 + 32.",
      });

    console.log(`  🔧 Tool calls: ${toolCalls?.length ?? 0}`);
    toolCalls?.forEach((tc) =>
      console.log(`     → ${tc.toolName}(${JSON.stringify(tc.args)})`),
    );
    console.log(`  📝 Response  :\n     ${text.replace(/\n/g, "\n     ")}`);
    console.log(`  🔄 Steps     : ${steps.length}`);
    console.log(`  🏁 Finish    : ${finishReason}`);
    console.log(`  ⏱️  Time      : ${timing(t)}`);

    const actualToolCalls = steps.flatMap((s) => s.toolCalls ?? []);
    if (actualToolCalls.length > 0)
      pass(
        `Tool calling — ${actualToolCalls.length} tool(s) invoked across ${steps.length} steps`,
      );
    else fail("Tool calling", "No tools were actually called");
  } catch (e) {
    fail("Tool calling", e.message);
  }
}

// ───────────────────────────────────────────────────────────────────────────
// 7. Multi-turn Chat — conversation history
// ───────────────────────────────────────────────────────────────────────────
async function test7_multiTurnChat() {
  header(7, "Multi-turn Chat — Conversation history");
  const t = Date.now();
  try {
    const turn1 = await generateText({
      model,
      messages: [
        {
          role: "user",
          content: "Tên tôi là Minh. Tôi thích lập trình Python.",
        },
      ],
    });
    console.log(`  👤 Turn 1 → : ${turn1.text.slice(0, 80).trim()}...`);

    const turn2 = await generateText({
      model,
      messages: [
        {
          role: "user",
          content: "Tên tôi là Minh. Tôi thích lập trình Python.",
        },
        { role: "assistant", content: turn1.text },
        {
          role: "user",
          content: "Bạn có nhớ tên tôi và ngôn ngữ tôi thích không?",
        },
      ],
    });
    console.log(`  👤 Turn 2 → : ${turn2.text.trim()}`);
    console.log(`  ⏱️  Time      : ${timing(t)}`);

    const txt = turn2.text.toLowerCase();
    if (txt.includes("minh") && txt.includes("python"))
      pass("Multi-turn — nhớ cả tên lẫn ngôn ngữ");
    else if (txt.includes("minh") || txt.includes("python"))
      pass("Multi-turn — nhớ một phần context");
    else fail("Multi-turn", "Không giữ được context cuộc hội thoại");
  } catch (e) {
    fail("Multi-turn", e.message);
  }
}

// ───────────────────────────────────────────────────────────────────────────
// 10. Error Handling
// ───────────────────────────────────────────────────────────────────────────
async function test10_errorHandling() {
  header(10, "Error Handling — Edge cases");

  // Case A: Model không tồn tại
  try {
    await generateText({
      model: ollamaProvider("non-existent-model-xyz"),
      prompt: "Hi",
    });
    fail("Wrong model", "Should have thrown");
  } catch (e) {
    pass(`Wrong model → throws correctly: "${e.message.slice(0, 55)}..."`);
  }

  // Case B: Zod schema mismatch (strict literal)
  try {
    const { object } = await generateObject({
      model,
      schema: z.object({
        price: z.number().min(1_000_000),
        currency: z.literal("VND"),
      }),
      prompt: "Giá một tô phở Hà Nội.",
      maxRetries: 1,
    });
    console.log(`  📦 schema got: ${JSON.stringify(object)}`);
    pass("Schema strict — model adapted correctly");
  } catch (e) {
    pass(`Schema mismatch handled: "${e.message.slice(0, 60)}"`);
  }

  // Case C: maxTokens = 1
  try {
    const { text, usage } = await generateText({
      model,
      prompt: "Kể về lịch sử Việt Nam.",
      maxTokens: 1,
    });
    pass(
      `maxTokens=1 — no crash, got: "${text.trim()}" (${usage.completionTokens} tok)`,
    );
  } catch (e) {
    pass(`maxTokens=1 throws gracefully: "${e.message.slice(0, 60)}"`);
  }

  // Case D: Empty prompt
  try {
    const { text } = await generateText({ model, prompt: "" });
    pass(`Empty prompt — model responded: "${text.trim().slice(0, 40)}..."`);
  } catch (e) {
    pass(`Empty prompt handled: "${e.message.slice(0, 60)}"`);
  }
}
